127class Frame:
128    def __init__(self, f, fmt):
129        self._f = f
130        self._fmt = fmt
131        channels = 3 if _top_level_pix_fmt(fmt["pix_fmt"]) == "rgb24" else 1
132        self.shape = (fmt["height"], fmt["width"], channels)
133
134        # denotes that the frame has not yet been modified
135        # when a frame is modified, it is converted to rgb24 first
136        self._modified = False
137
138    def _mut(self):
139        if self._modified:
140            assert self._fmt["pix_fmt"] in ["rgb24", "gray"]
141            return
142
143        self._modified = True
144        if (
145            self._fmt["pix_fmt"] != "rgb24"
146            and _top_level_pix_fmt(self._fmt["pix_fmt"]) == "rgb24"
147        ):
148            self._f = _filter_scale(self._f, pix_fmt="rgb24")
149            self._fmt["pix_fmt"] = "rgb24"
150        elif (
151            self._fmt["pix_fmt"] != "gray"
152            and _top_level_pix_fmt(self._fmt["pix_fmt"]) == "gray"
153        ):
154            self._f = _filter_scale(self._f, pix_fmt="gray")
155            self._fmt["pix_fmt"] = "gray"
156
157    def copy(self):
158        return Frame(self._f, self._fmt.copy())
159
160    def numpy(self):
161        """
162        Return the frame as a numpy array.
163        """
164
165        self._mut()
166        server = _server()
167        frame = server.frame(
168            self.shape[1], self.shape[0], self._fmt["pix_fmt"], self._f
169        )
170        assert type(frame) is bytes
171        assert len(frame) == self.shape[0] * self.shape[1] * self.shape[2]
172        raw_data_array = np.frombuffer(frame, dtype=np.uint8)
173        frame = raw_data_array.reshape(self.shape)
174        if self.shape[2] == 3:
175            frame = frame[:, :, ::-1]  # convert RGB to BGR
176        return frame
177
178    def __getitem__(self, key):
179        if not isinstance(key, tuple):
180            raise NotImplementedError("Only 2D slicing is supported")
181
182        if len(key) != 2:
183            raise NotImplementedError("Only 2D slicing is supported")
184
185        if not all(isinstance(x, slice) for x in key):
186            raise NotImplementedError("Only 2D slicing is supported")
187
188        miny = key[0].start if key[0].start is not None else 0
189        maxy = key[0].stop if key[0].stop is not None else self.shape[0]
190        minx = key[1].start if key[1].start is not None else 0
191        maxx = key[1].stop if key[1].stop is not None else self.shape[1]
192
193        # handle negative indices
194        if miny < 0:
195            miny = self.shape[0] + miny
196        if maxy < 0:
197            maxy = self.shape[0] + maxy
198        if minx < 0:
199            minx = self.shape[1] + minx
200        if maxx < 0:
201            maxx = self.shape[1] + maxx
202
203        if (
204            maxy <= miny
205            or maxx <= minx
206            or miny < 0
207            or minx < 0
208            or maxy > self.shape[0]
209            or maxx > self.shape[1]
210        ):
211            raise NotImplementedError("Invalid slice")
212
213        f = _slice_mat(self._f, miny, maxy, minx, maxx)
214        fmt = self._fmt.copy()
215        fmt["width"] = maxx - minx
216        fmt["height"] = maxy - miny
217        return Frame(f, fmt)
218
219    def __setitem__(self, key, value):
220        if type(key) is tuple:
221            value = frameify(value, "value")
222
223            if len(key) != 2:
224                raise NotImplementedError("Only 2D slicing is supported")
225
226            if not all(isinstance(x, slice) for x in key):
227                raise NotImplementedError("Only 2D slicing is supported")
228
229            miny = key[0].start if key[0].start is not None else 0
230            maxy = key[0].stop if key[0].stop is not None else self.shape[0]
231            minx = key[1].start if key[1].start is not None else 0
232            maxx = key[1].stop if key[1].stop is not None else self.shape[1]
233
234            # handle negative indices
235            if miny < 0:
236                miny = self.shape[0] + miny
237            if maxy < 0:
238                maxy = self.shape[0] + maxy
239            if minx < 0:
240                minx = self.shape[1] + minx
241            if maxx < 0:
242                maxx = self.shape[1] + maxx
243
244            if (
245                maxy <= miny
246                or maxx <= minx
247                or miny < 0
248                or minx < 0
249                or maxy > self.shape[0]
250                or maxx > self.shape[1]
251            ):
252                raise NotImplementedError("Invalid slice")
253
254            if value.shape[0] != maxy - miny or value.shape[1] != maxx - minx:
255                raise NotImplementedError("Shape mismatch")
256
257            self._mut()
258            value._mut()
259
260            self._f = _slice_write_mat(self._f, value._f, miny, maxy, minx, maxx)
261        elif type(key) is Frame or type(key) is np.ndarray:
262            key = frameify(key, "key")
263
264            if key.shape[0] != self.shape[0] or key.shape[1] != self.shape[1]:
265                raise NotImplementedError("Shape mismatch")
266
267            if key.shape[2] != 1:
268                raise NotImplementedError("Only 1-channel mask frames are supported")
269
270            # Value should be a bgr or bgra color
271            if (type(value) is not list and type(value) is not tuple) or len(
272                value
273            ) not in [3, 4]:
274                raise NotImplementedError(
275                    "Value should be a 3 or 4 element list or tuple"
276                )
277            value = [float(x) for x in value]
278            if len(value) == 3:
279                value.append(255.0)
280
281            self._mut()
282            key._mut()
283
284            self._f = _set_to(self._f, value, key._f)
285        else:
286            raise NotImplementedError(
287                "__setitem__ only supports slicing by a 2d tuple or a mask frame"
288            )

291def _inline_frame(arr):
292    if arr.dtype != np.uint8:
293        raise Exception("Only uint8 arrays are supported")
294    if len(arr.shape) != 3:
295        raise Exception("Only 3D arrays are supported")
296    if arr.shape[2] != 3:
297        raise Exception("To inline a frame, the array must have 3 channels")
298
299    arr = arr[:, :, ::-1]
300    if not arr.flags["C_CONTIGUOUS"]:
301        arr = np.ascontiguousarray(arr)
302
303    width = arr.shape[1]
304    height = arr.shape[0]
305    pix_fmt = "rgb24"
306
307    data_gzip = zlib.compress(memoryview(arr), level=1)
308
309    f = _inline_mat(
310        data_gzip, width=width, height=height, pix_fmt=pix_fmt, compression="zlib"
311    )
312    fmt = {"width": width, "height": height, "pix_fmt": pix_fmt}
313
314    # Return the resulting Frame object
315    return Frame(f, fmt)

325class VideoCapture:
326    def __init__(self, path: str):
327        server = _server()
328        if type(path) is str:
329            match = re.match(r"(http|https)://([^/]+)(.*)", path)
330            if match is not None:
331                endpoint = f"{match.group(1)}://{match.group(2)}"
332                path = match.group(3)
333                if path.startswith("/"):
334                    path = path[1:]
335                self._path = path
336                self._source = server.source(path, 0, "http", {"endpoint": endpoint})
337            else:
338                self._path = path
339                self._source = server.source(path, 0, "fs", {"root": "."})
340        elif isinstance(path, vf.Source):
341            assert isinstance(server, vf.Server)
342            self._path = path._name
343            self._source = path
344        self._next_frame_idx = 0
345
346    def isOpened(self) -> bool:
347        return True
348
349    def get(self, prop):
350        if prop == CAP_PROP_FPS:
351            return _ts_to_fps(self._source.ts())
352        elif prop == CAP_PROP_FRAME_WIDTH:
353            return self._source.fmt()["width"]
354        elif prop == CAP_PROP_FRAME_HEIGHT:
355            return self._source.fmt()["height"]
356        elif prop == CAP_PROP_FRAME_COUNT:
357            return len(self._source)
358        elif prop == CAP_PROP_POS_FRAMES:
359            return self._next_frame_idx
360
361        raise Exception(f"Unknown property {prop}")
362
363    def set(self, prop, value):
364        if prop == CAP_PROP_POS_FRAMES:
365            assert value >= 0 and value < len(self._source.ts())
366            self._next_frame_idx = value
367        elif prop == CAP_PROP_POS_MSEC:
368            t = Fraction(value, 1000)
369            ts = self._source.ts()
370            next_frame_idx = bisect_right(ts, t)
371            self._next_frame_idx = next_frame_idx
372        else:
373            raise Exception(f"Unsupported property {prop}")
374
375    def read(self):
376        if self._next_frame_idx >= len(self._source):
377            return False, None
378        frame = self._source.iloc[self._next_frame_idx]
379        self._next_frame_idx += 1
380        frame = Frame(frame, self._source.fmt())
381        return True, frame
382
383    def __getitem__(self, key):
384        if not isinstance(key, int):
385            raise NotImplementedError("Only integer indexing is supported")
386        if key < 0:
387            key = len(self._source) + key
388        if key < 0 or key >= len(self._source):
389            raise IndexError("Index out of bounds")
390        frame = self._source.iloc[key]
391        frame = Frame(frame, self._source.fmt())
392        return frame
393
394    def __len__(self):
395        return len(self._source)
396
397    def release(self):
398        pass

401class VideoWriter:
402    def __init__(
403        self,
404        path,
405        fourcc,
406        fps,
407        size,
408        batch_size=1024,
409        compression="gzip",
410        ttl=3600,
411        pix_fmt="yuv420p",
412        vod_segment_length=Fraction(2, 1),
413    ):
414        server = _server()
415        assert isinstance(server, vf.Server)
416        assert path is None or type(path) is str
417        self._path = path
418        if isinstance(fps, int):
419            self._f_time = Fraction(1, fps)
420        elif isinstance(fps, Fraction):
421            self._f_time = 1 / fps
422        else:
423            raise Exception("fps must be an integer or a Fraction")
424
425        assert isinstance(size, tuple) or isinstance(size, list)
426        assert len(size) == 2
427        width, height = size
428        assert ttl is None or isinstance(ttl, int)
429        self._spec = server.create_spec(
430            width, height, pix_fmt, vod_segment_length, 1 / self._f_time, ttl=ttl
431        )
432        self._batch_size = batch_size
433        assert compression is None or compression in ["gzip"]
434        self._compression = compression
435        self._idx = 0
436        self._feb = vf._FrameExpressionBlock()
437
438    def _flush(self, terminal=False):
439        server = _server()
440        if len(self._feb) > 0:
441            server.push_spec_part_block(
442                self._spec,
443                self._idx - len(self._feb),
444                [self._feb],
445                terminal=terminal,
446                compression=self._compression,
447            )
448            self._feb = vf._FrameExpressionBlock()
449        else:
450            server.push_spec_part_block(
451                self._spec,
452                self._idx - len(self._feb),
453                [],
454                terminal=terminal,
455            )
456
457    def spec(self):
458        return self._spec
459
460    def write(self, frame):
461        if frame is not None:
462            frame = frameify(frame, "frame")
463            if frame._fmt["width"] != self._spec._fmt["width"]:
464                raise Exception(
465                    f"Frame type error; expected width {self._spec._fmt['width']}, got {frame._fmt['width']}"
466                )
467            if frame._fmt["height"] != self._spec._fmt["height"]:
468                raise Exception(
469                    f"Frame type error; expected height {self._spec._fmt['height']}, got {frame._fmt['height']}"
470                )
471            if frame._fmt["pix_fmt"] != self._spec._fmt["pix_fmt"]:
472                f_obj = _filter_scale(frame._f, pix_fmt=self._spec._fmt["pix_fmt"])
473                frame = Frame(f_obj, self._spec._fmt)
474        self._feb.insert_frame(frame._f if frame is not None else None)
475        self._idx += 1
476
477        if len(self._feb) >= self._batch_size:
478            self._flush()
479
480    def release(self):
481        self._flush(True)
482        if self._path is not None:
483            server = _server()
484            server.export_spec(self._spec.id(), self._path)

525def imwrite(path, img, *args):
526    if len(args) > 0:
527        raise NotImplementedError("imwrite does not support additional arguments")
528
529    img = frameify(img)
530    fmt = img._fmt.copy()
531    width = fmt["width"]
532    height = fmt["height"]
533
534    if path.lower().endswith(".png"):
535        out_pix_fmt = "rgb24"
536        encoder = "png"
537    elif path.lower().endswith((".jpg", ".jpeg")):
538        encoder = "mjpeg"
539        if img._fmt["pix_fmt"] not in ["yuvj420p", "yuvj422p", "yuvj444p"]:
540            out_pix_fmt = "yuvj420p"
541        else:
542            out_pix_fmt = img._fmt["pix_fmt"]
543    else:
544        raise Exception("Unsupported image format")
545
546    if img._fmt["pix_fmt"] != out_pix_fmt:
547        f = _filter_scale(img._f, pix_fmt=out_pix_fmt)
548        img = Frame(f, {"width": width, "height": height, "pix_fmt": out_pix_fmt})
549
550    writer = VideoWriter(None, None, 1, (width, height), pix_fmt=out_pix_fmt)
551    writer.write(img)
552    writer.release()
553
554    spec = writer.spec()
555    server = _server()
556    server.export_spec(spec.id(), path, encoder=encoder)

 111def _ts_to_fps(timestamps):
 112    if len(timestamps) < 2:
 113        return 0
 114    fps = Fraction(len(timestamps), timestamps[-1] - timestamps[0])
 115    if fps.denominator == 1:
 116        return fps.numerator
 117    return float(fps)

 175class Frame:
 176    def __init__(self, f, fmt, parent=None, slice_bounds=None):
 177        self._f_internal = f
 178        self._fmt = fmt
 179        channels = 3 if _top_level_pix_fmt(fmt["pix_fmt"]) == "rgb24" else 1
 180        self.shape = (fmt["height"], fmt["width"], channels)
 181
 182        # denotes that the frame has not yet been modified
 183        # when a frame is modified, it is converted to rgb24 first
 184        self._modified = False
 185
 186        # For slice views: track parent frame and bounds for write-back
 187        self._parent = parent
 188        self._slice_bounds = slice_bounds  # (miny, maxy, minx, maxx)
 189
 190    @property
 191    def _f(self):
 192        return self._f_internal
 193
 194    @_f.setter
 195    def _f(self, value):
 196        self._f_internal = value
 197        # If this is a slice view, propagate changes back to parent
 198        if self._parent is not None:
 199            miny, maxy, minx, maxx = self._slice_bounds
 200            self._parent._mut()
 201            self._parent._f = _slice_write_mat(
 202                self._parent._f, self._f_internal, miny, maxy, minx, maxx
 203            )
 204
 205    def _mut(self):
 206        if self._modified:
 207            assert self._fmt["pix_fmt"] in ["rgb24", "gray"]
 208            return
 209
 210        self._modified = True
 211        if (
 212            self._fmt["pix_fmt"] != "rgb24"
 213            and _top_level_pix_fmt(self._fmt["pix_fmt"]) == "rgb24"
 214        ):
 215            self._f = _filter_scale(self._f, pix_fmt="rgb24")
 216            self._fmt["pix_fmt"] = "rgb24"
 217        elif (
 218            self._fmt["pix_fmt"] != "gray"
 219            and _top_level_pix_fmt(self._fmt["pix_fmt"]) == "gray"
 220        ):
 221            self._f = _filter_scale(self._f, pix_fmt="gray")
 222            self._fmt["pix_fmt"] = "gray"
 223
 224    def copy(self):
 225        # Copy creates an independent frame (no parent reference)
 226        return Frame(self._f, self._fmt.copy())
 227
 228    def numpy(self):
 229        """
 230        Return the frame as a numpy array.
 231        """
 232
 233        self._mut()
 234        server = _server()
 235        frame = server.frame(
 236            self.shape[1], self.shape[0], self._fmt["pix_fmt"], self._f
 237        )
 238        assert type(frame) is bytes
 239        assert len(frame) == self.shape[0] * self.shape[1] * self.shape[2]
 240        raw_data_array = np.frombuffer(frame, dtype=np.uint8)
 241        frame = raw_data_array.reshape(self.shape)
 242        if self.shape[2] == 3:
 243            frame = frame[:, :, ::-1]  # convert RGB to BGR
 244        return frame
 245
 246    def __getitem__(self, key):
 247        if not isinstance(key, tuple):
 248            raise NotImplementedError("Only 2D slicing is supported")
 249
 250        if len(key) != 2:
 251            raise NotImplementedError("Only 2D slicing is supported")
 252
 253        if not all(isinstance(x, slice) for x in key):
 254            raise NotImplementedError("Only 2D slicing is supported")
 255
 256        miny = key[0].start if key[0].start is not None else 0
 257        maxy = key[0].stop if key[0].stop is not None else self.shape[0]
 258        minx = key[1].start if key[1].start is not None else 0
 259        maxx = key[1].stop if key[1].stop is not None else self.shape[1]
 260
 261        # handle negative indices
 262        if miny < 0:
 263            miny = self.shape[0] + miny
 264        if maxy < 0:
 265            maxy = self.shape[0] + maxy
 266        if minx < 0:
 267            minx = self.shape[1] + minx
 268        if maxx < 0:
 269            maxx = self.shape[1] + maxx
 270
 271        if (
 272            maxy <= miny
 273            or maxx <= minx
 274            or miny < 0
 275            or minx < 0
 276            or maxy > self.shape[0]
 277            or maxx > self.shape[1]
 278        ):
 279            raise NotImplementedError("Invalid slice")
 280
 281        f = _slice_mat(self._f, miny, maxy, minx, maxx)
 282        fmt = self._fmt.copy()
 283        fmt["width"] = maxx - minx
 284        fmt["height"] = maxy - miny
 285        # Create slice with parent reference for write-back propagation
 286        return Frame(f, fmt, parent=self, slice_bounds=(miny, maxy, minx, maxx))
 287
 288    def __setitem__(self, key, value):
 289        if type(key) is tuple:
 290            if len(key) != 2:
 291                raise NotImplementedError("Only 2D slicing is supported")
 292
 293            if not all(isinstance(x, slice) for x in key):
 294                raise NotImplementedError("Only 2D slicing is supported")
 295
 296            miny = key[0].start if key[0].start is not None else 0
 297            maxy = key[0].stop if key[0].stop is not None else self.shape[0]
 298            minx = key[1].start if key[1].start is not None else 0
 299            maxx = key[1].stop if key[1].stop is not None else self.shape[1]
 300
 301            # handle negative indices
 302            if miny < 0:
 303                miny = self.shape[0] + miny
 304            if maxy < 0:
 305                maxy = self.shape[0] + maxy
 306            if minx < 0:
 307                minx = self.shape[1] + minx
 308            if maxx < 0:
 309                maxx = self.shape[1] + maxx
 310
 311            if (
 312                maxy <= miny
 313                or maxx <= minx
 314                or miny < 0
 315                or minx < 0
 316                or maxy > self.shape[0]
 317                or maxx > self.shape[1]
 318            ):
 319                raise NotImplementedError("Invalid slice")
 320
 321            # Check if value is a color tuple/list or scalar (solid fill)
 322            if isinstance(value, (int, float)):
 323                # Scalar value - treat as grayscale
 324                height = maxy - miny
 325                width = maxx - minx
 326                v = int(value)
 327                value = solid((height, width, self.shape[2]), (v, v, v))
 328            elif isinstance(value, (tuple, list)) and len(value) in [3, 4]:
 329                # Create a solid color frame of the right size
 330                height = maxy - miny
 331                width = maxx - minx
 332                value = solid((height, width, self.shape[2]), value[:3])
 333            else:
 334                value = frameify(value, "value")
 335
 336            if value.shape[0] != maxy - miny or value.shape[1] != maxx - minx:
 337                raise NotImplementedError("Shape mismatch")
 338
 339            self._mut()
 340            value._mut()
 341
 342            self._f = _slice_write_mat(self._f, value._f, miny, maxy, minx, maxx)
 343        elif type(key) is Frame or type(key) is np.ndarray:
 344            key = frameify(key, "key")
 345
 346            if key.shape[0] != self.shape[0] or key.shape[1] != self.shape[1]:
 347                raise NotImplementedError("Shape mismatch")
 348
 349            if key.shape[2] != 1:
 350                raise NotImplementedError("Only 1-channel mask frames are supported")
 351
 352            # Value should be a bgr or bgra color
 353            if (type(value) is not list and type(value) is not tuple) or len(
 354                value
 355            ) not in [3, 4]:
 356                raise NotImplementedError(
 357                    "Value should be a 3 or 4 element list or tuple"
 358                )
 359            value = [float(x) for x in value]
 360            if len(value) == 3:
 361                value.append(255.0)
 362
 363            self._mut()
 364            key._mut()
 365
 366            self._f = _set_to(self._f, value, key._f)
 367        else:
 368            raise NotImplementedError(
 369                "__setitem__ only supports slicing by a 2d tuple or a mask frame"
 370            )

 373def _inline_frame(arr):
 374    if arr.dtype != np.uint8:
 375        raise Exception("Only uint8 arrays are supported")
 376    if len(arr.shape) != 3:
 377        raise Exception("Only 3D arrays are supported")
 378    if arr.shape[2] != 3:
 379        raise Exception("To inline a frame, the array must have 3 channels")
 380
 381    arr = arr[:, :, ::-1]
 382    if not arr.flags["C_CONTIGUOUS"]:
 383        arr = np.ascontiguousarray(arr)
 384
 385    width = arr.shape[1]
 386    height = arr.shape[0]
 387    pix_fmt = "rgb24"
 388
 389    data_gzip = zlib.compress(memoryview(arr), level=1)
 390
 391    f = _inline_mat(
 392        data_gzip, width=width, height=height, pix_fmt=pix_fmt, compression="zlib"
 393    )
 394    fmt = {"width": width, "height": height, "pix_fmt": pix_fmt}
 395
 396    # Return the resulting Frame object
 397    return Frame(f, fmt)

 407class VideoCapture:
 408    def __init__(self, path: str):
 409        server = _server()
 410        if type(path) is str:
 411            match = re.match(r"(http|https)://([^/]+)(.*)", path)
 412            if match is not None:
 413                endpoint = f"{match.group(1)}://{match.group(2)}"
 414                path = unquote(match.group(3))
 415                if path.startswith("/"):
 416                    path = path[1:]
 417                self._path = path
 418                self._source = server.source(path, 0, "http", {"endpoint": endpoint})
 419            else:
 420                self._path = path
 421                self._source = server.source(path, 0, "fs", {"root": "."})
 422        elif isinstance(path, vf.Source):
 423            assert isinstance(server, vf.Server)
 424            self._path = path._name
 425            self._source = path
 426        self._next_frame_idx = 0
 427
 428    def isOpened(self) -> bool:
 429        return True
 430
 431    def get(self, prop):
 432        if prop == CAP_PROP_FPS:
 433            return _ts_to_fps(self._source.ts())
 434        elif prop == CAP_PROP_FRAME_WIDTH:
 435            return self._source.fmt()["width"]
 436        elif prop == CAP_PROP_FRAME_HEIGHT:
 437            return self._source.fmt()["height"]
 438        elif prop == CAP_PROP_FRAME_COUNT:
 439            return len(self._source)
 440        elif prop == CAP_PROP_POS_FRAMES:
 441            return self._next_frame_idx
 442        elif prop == CAP_PROP_POS_MSEC:
 443            ts = self._source.ts()
 444            if self._next_frame_idx >= len(ts):
 445                # Past the end, return the last timestamp
 446                if len(ts) > 0:
 447                    return float(ts[-1] * 1000)
 448                return 0.0
 449            return float(ts[self._next_frame_idx] * 1000)
 450
 451        raise Exception(f"Unknown property {prop}")
 452
 453    def set(self, prop, value):
 454        if prop == CAP_PROP_POS_FRAMES:
 455            assert value >= 0 and value < len(self._source)
 456            self._next_frame_idx = value
 457        elif prop == CAP_PROP_POS_MSEC:
 458            t = Fraction(int(value), 1000)
 459            ts = self._source.ts()
 460            next_frame_idx = bisect_right(ts, t)
 461            self._next_frame_idx = next_frame_idx
 462        else:
 463            raise Exception(f"Unsupported property {prop}")
 464
 465    def read(self):
 466        if self._next_frame_idx >= len(self._source):
 467            return False, None
 468        frame = self._source.iloc[self._next_frame_idx]
 469        self._next_frame_idx += 1
 470        frame = Frame(frame, self._source.fmt())
 471        return True, frame
 472
 473    def __getitem__(self, key):
 474        if not isinstance(key, int):
 475            raise NotImplementedError("Only integer indexing is supported")
 476        if key < 0:
 477            key = len(self._source) + key
 478        if key < 0 or key >= len(self._source):
 479            raise IndexError("Index out of bounds")
 480        frame = self._source.iloc[key]
 481        frame = Frame(frame, self._source.fmt())
 482        return frame
 483
 484    def __len__(self):
 485        return len(self._source)
 486
 487    def release(self):
 488        pass

 491class VideoWriter:
 492    def __init__(
 493        self,
 494        path,
 495        fourcc,
 496        fps,
 497        size,
 498        batch_size=1024,
 499        initial_batch_size=100,
 500        compression="gzip",
 501        ttl=3600,
 502        pix_fmt="yuv420p",
 503        vod_segment_length=Fraction(2, 1),
 504    ):
 505        server = _server()
 506        assert isinstance(server, vf.Server)
 507        assert path is None or type(path) is str
 508        if path is not None and server.is_vod_only():
 509            path = None
 510
 511        self._path = path
 512        if isinstance(fps, int):
 513            self._f_time = Fraction(1, fps)
 514        elif isinstance(fps, Fraction):
 515            self._f_time = 1 / fps
 516        elif isinstance(fps, float):
 517            # 29.97
 518            if abs(fps - 30000 / 1001) < 1e-6:
 519                self._f_time = Fraction(1001, 30000)
 520            # 23.976
 521            elif abs(fps - 24000 / 1001) < 1e-6:
 522                self._f_time = Fraction(1001, 24000)
 523            # 59.94
 524            elif abs(fps - 60000 / 1001) < 1e-6:
 525                self._f_time = Fraction(1001, 60000)
 526            else:
 527                # Round to nearest integer fps
 528                self._f_time = Fraction(1, int(round(fps)))
 529        else:
 530            raise Exception("fps must be an integer, float, or Fraction")
 531
 532        assert isinstance(size, tuple) or isinstance(size, list)
 533        assert len(size) == 2
 534        width, height = size
 535        assert ttl is None or isinstance(ttl, int)
 536        self._spec = server.create_spec(
 537            width, height, pix_fmt, vod_segment_length, 1 / self._f_time, ttl=ttl
 538        )
 539        self._batch_size = batch_size
 540        self._initial_batch_size = initial_batch_size
 541        assert compression is None or compression in ["gzip"]
 542        self._compression = compression
 543        self._idx = 0
 544        self._feb = vf._FrameExpressionBlock()
 545        self._first_push_done = False
 546
 547        # writer_init_callback
 548        if server.cv2_writer_init_callback() is not None:
 549            server.cv2_writer_init_callback()(self)
 550
 551    def _flush(self, terminal=False):
 552        server = _server()
 553        if len(self._feb) > 0:
 554            server.push_spec_part_block(
 555                self._spec,
 556                self._idx - len(self._feb),
 557                [self._feb],
 558                terminal=terminal,
 559                compression=self._compression,
 560            )
 561            self._feb = vf._FrameExpressionBlock()
 562        else:
 563            server.push_spec_part_block(
 564                self._spec,
 565                self._idx - len(self._feb),
 566                [],
 567                terminal=terminal,
 568            )
 569
 570        # First push callback (called once after first successful push)
 571        if not self._first_push_done:
 572            self._first_push_done = True

 574                server.cv2_writer_first_push_callback()(self)

 583                raise Exception(
 584                    f"Frame type error; expected width {self._spec._fmt['width']}, got {frame._fmt['width']}"
 585                )

 587                raise Exception(
 588                    f"Frame type error; expected height {self._spec._fmt['height']}, got {frame._fmt['height']}"
 589                )
 590            if frame._fmt["pix_fmt"] != self._spec._fmt["pix_fmt"]:
 591                f_obj = _filter_scale(frame._f, pix_fmt=self._spec._fmt["pix_fmt"])
 592                frame = Frame(f_obj, self._spec._fmt)
 593        self._feb.insert_frame(frame._f if frame is not None else None)
 594        self._idx += 1
 595
 596        # Use smaller initial batch for faster first response
 597        threshold = (
 598            self._initial_batch_size if not self._first_push_done else self._batch_size
 599        )
 600        if len(self._feb) >= threshold:
 601            self._flush()

 604        return True

 607        self._flush(True)
 608        if self._path is not None:

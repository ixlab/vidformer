vidformer.numpy

vidformer.numpy mimics numpy's API while supporting vidformer Frames.

When Frames are detected, dispatches to cv2 functions. Otherwise falls back to numpy.

  2vidformer.numpy mimics numpy's API while supporting vidformer Frames.
  3
  4When Frames are detected, dispatches to cv2 functions. Otherwise falls back to numpy.

 18def _is_frame(obj):
 19    return isinstance(obj, _vf_cv2.Frame)

 22def _contains_frame(iterable):
 23    return any(_is_frame(x) for x in iterable)

 26def _is_frame_shape(shape, dtype):
 27    """Check if shape/dtype combination should produce a Frame."""
 28    return (
 29        isinstance(shape, (tuple, list))
 30        and len(shape) == 3
 31        and shape[2] in (1, 3)
 32        and (dtype is None or dtype == _np.uint8)
 33    )

 36def zeros(shape, dtype=None, *args, **kwargs):
 37    """Mimics numpy.zeros. Returns a Frame for video-like shapes (h, w, 1|3) with uint8."""
 38    if not args and not kwargs and _is_frame_shape(shape, dtype):
 39        return _vf_cv2.zeros(shape, dtype=_np.uint8)
 40
 41    if dtype is None:
 42        return _np.zeros(shape, *args, **kwargs)
 43    return _np.zeros(shape, dtype=dtype, *args, **kwargs)
 44
 45
 46def ones(shape, dtype=None, *args, **kwargs):
 47    """Mimics numpy.ones. Returns a Frame for video-like shapes (h, w, 1|3) with uint8."""
 48    if not args and not kwargs and _is_frame_shape(shape, dtype):
 49        return _vf_cv2.solid(shape, (1, 1, 1), dtype=_np.uint8)
 50
 51    if dtype is None:
 52        return _np.ones(shape, *args, **kwargs)
 53    return _np.ones(shape, dtype=dtype, *args, **kwargs)

 56def _parse_fill_value(fill_value):
 57    """Convert fill_value to a color tuple, or return None if unsupported."""
 58    if isinstance(fill_value, (int, float)):
 59        v = int(fill_value)
 60        return (v, v, v)
 61    elif isinstance(fill_value, (tuple, list)) and len(fill_value) == 3:
 62        return tuple(int(c) for c in fill_value)
 63    return None

 66def full(shape, fill_value, dtype=None, *args, **kwargs):
 67    """Mimics numpy.full. Returns a Frame for video-like shapes (h, w, 1|3) with uint8."""
 68    if not args and not kwargs and _is_frame_shape(shape, dtype):
 69        color = _parse_fill_value(fill_value)
 70        if color is not None:
 71            return _vf_cv2.solid(shape, color, dtype=_np.uint8)
 72
 73    if dtype is None:
 74        return _np.full(shape, fill_value, *args, **kwargs)
 75    return _np.full(shape, fill_value, dtype=dtype, *args, **kwargs)
 76

 78def zeros_like(a, dtype=None, *args, **kwargs):
 79    """Mimics numpy.zeros_like. Returns a Frame if input is a Frame."""
 80    if _is_frame(a):
 81        return _vf_cv2.zeros(a.shape, dtype=_np.uint8)
 82    return _np.zeros_like(a, dtype=dtype, *args, **kwargs)

 92def full_like(a, fill_value, dtype=None, *args, **kwargs):
 93    """Mimics numpy.full_like. Returns a Frame if input is a Frame."""
 94    if _is_frame(a):
 95        color = _parse_fill_value(fill_value)
 96        if color is not None:

 98        # Fall back to numpy for unsupported fill_value types
 99        return _np.full_like(a.numpy(), fill_value, dtype=dtype, *args, **kwargs)
100    return _np.full_like(a, fill_value, dtype=dtype, *args, **kwargs)

117def concatenate(arrays, axis=0, *args, **kwargs):
118    """Mimics numpy.concatenate. Uses cv2.hconcat/vconcat for Frames."""
119    if _contains_frame(arrays):
120        if axis == 0:
121            return _vf_cv2.vconcat(arrays)
122        elif axis == 1:
123            return _vf_cv2.hconcat(arrays)
124        else:
125            raise ValueError(

127            )
128    return _np.concatenate(arrays, axis=axis, *args, **kwargs)

131def flip(m, axis=None, *args, **kwargs):
132    """Mimics numpy.flip. Uses cv2.flip for Frames."""
133    if _is_frame(m):
134        if axis is None:
135            return _vf_cv2.flip(m, -1)
136        elif axis == 0:
137            return _vf_cv2.flip(m, 0)
138        elif axis == 1:
139            return _vf_cv2.flip(m, 1)
140        else:
141            raise ValueError(
142                f"Unsupported axis {axis} for Frame flip (only 0, 1, or None)"
143            )
144    return _np.flip(m, axis=axis, *args, **kwargs)

147def rot90(m, k=1, axes=(0, 1), *args, **kwargs):
148    """Mimics numpy.rot90. Uses cv2.rotate for Frames."""
149    if _is_frame(m):
150        if axes != (0, 1):
151            raise ValueError("Only axes=(0, 1) supported for Frame rotation")
152        k = k % 4
153        if k == 0:
154            return m
155        elif k == 1:
156            return _vf_cv2.rotate(m, _vf_cv2.ROTATE_90_COUNTERCLOCKWISE)
157        elif k == 2:
158            return _vf_cv2.rotate(m, _vf_cv2.ROTATE_180)
159        elif k == 3:
160            return _vf_cv2.rotate(m, _vf_cv2.ROTATE_90_CLOCKWISE)
161    return _np.rot90(m, k=k, axes=axes, *args, **kwargs)

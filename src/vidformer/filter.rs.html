16/// A decoded video frame
17#[derive(Clone)]
18pub struct Frame {
19    inner: Arc<AVFrame>,
20    pub width: i32,
21    pub height: i32,
22    pub format: ffi::AVPixelFormat,
23}
24
25impl Frame {
26    pub(crate) fn new(inner: AVFrame) -> Self {
27        let width = unsafe { (*inner.inner).width };
28        let height = unsafe { (*inner.inner).height };
29        let format = unsafe { (*inner.inner).format };
30        Frame {
31            inner: Arc::new(inner),
32            width,
33            height,
34            format,
35        }
36    }
37
38    pub(crate) fn new_arc(inner: Arc<AVFrame>) -> Self {
39        let width = unsafe { (*inner.inner).width };
40        let height = unsafe { (*inner.inner).height };
41        let format = unsafe { (*inner.inner).format };
42        Frame {
43            inner,
44            width,
45            height,
46            format,
47        }
48    }
49
50    pub(crate) fn into_avframe(self) -> Arc<AVFrame> {
51        self.inner
52    }
53}

65/// A value that can be passed to a filter
66///
67/// This can be a either a video frame or a conventional data value
68#[derive(Debug, Clone)]
69pub enum Val {
70    Frame(Frame),
71    FrameType(FrameType),
72    Bool(bool),
73    Int(i64),
74    String(String),
75    Bytes(Vec<u8>),
76    Float(f64),
77    List(Vec<Val>),
78}

80#[derive(PartialEq, Eq, Debug, Clone)]
81pub struct FrameType {

83    pub height: usize,
84    pub format: ffi::AVPixelFormat,
85}

87impl FrameType {

89        FrameType {

91            height,

93        }
94    }

97impl Val {
98    pub fn from_data_expr_with_frame_converter<F>(
99        expr: &crate::sir::DataExpr,
100        _context: &crate::dve::Context,
101        frame_converter: F,
102    ) -> Self
103    where
104        F: Fn(&crate::sir::FrameExpr) -> Val + Copy,
105    {
106        match expr {
107            crate::sir::DataExpr::Bool(b) => Val::Bool(*b),
108            crate::sir::DataExpr::Int(i) => Val::Int(*i),
109            crate::sir::DataExpr::String(s) => Val::String(s.to_string()),
110            crate::sir::DataExpr::Bytes(b) => Val::Bytes(b.clone()),
111            crate::sir::DataExpr::Float(f) => Val::Float(*f),
112            crate::sir::DataExpr::List(list) => {
113                let list = list
114                    .iter()
115                    .map(|item| match item {
116                        crate::sir::Expr::Frame(frame) => frame_converter(frame),
117                        crate::sir::Expr::Data(data) => Self::from_data_expr_with_frame_converter(
118                            data,
119                            _context,
120                            frame_converter,
121                        ),
122                    })
123                    .collect();
124                Val::List(list)
125            }
126        }
127    }
128
129    pub(crate) fn as_int(&self) -> Option<i64> {
130        match self {
131            Val::Int(i) => Some(*i),
132            _ => None,
133        }
134    }
135}

137/// A filter that can create a video frame from some inputs
138pub trait Filter: Send + Sync {
139    /// Creates a video frame from some inputs
140    ///
141    /// The inputs are arbitrary and user provided. An input can either be
142    /// ordered (arg) or named (kwargs). This is similar to Python function calls.
143    fn filter(
144        &self,
145        args: &[Val],
146        kwargs: &BTreeMap<String, Val>,
147    ) -> Result<Frame, crate::dve::Error>;
148
149    fn filter_type(
150        &self,
151        args: &[Val],
152        kwargs: &BTreeMap<String, Val>,
153    ) -> Result<FrameType, crate::dve::Error>;
154}

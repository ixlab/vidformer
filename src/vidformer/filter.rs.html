16/// A decoded video frame
17#[derive(Clone)]
18pub struct Frame {
19    inner: Arc<AVFrame>,
20    pub width: i32,
21    pub height: i32,
22    pub format: ffi::AVPixelFormat,
23}
24
25impl Frame {
26    pub(crate) fn new(inner: AVFrame) -> Self {
27        let width = unsafe { (*inner.inner).width };
28        let height = unsafe { (*inner.inner).height };
29        let format = unsafe { (*inner.inner).format };
30        Frame {
31            inner: Arc::new(inner),
32            width,
33            height,
34            format,
35        }
36    }
37
38    pub(crate) fn new_arc(inner: Arc<AVFrame>) -> Self {
39        let width = unsafe { (*inner.inner).width };
40        let height = unsafe { (*inner.inner).height };
41        let format = unsafe { (*inner.inner).format };
42        Frame {
43            inner,
44            width,
45            height,
46            format,
47        }
48    }
49
50    pub(crate) fn into_avframe(self) -> Arc<AVFrame> {
51        self.inner
52    }
53}

55impl std::fmt::Debug for Frame {

65/// A value that can be passed to a filter
66///
67/// This can be a either a video frame or a conventional data value
68#[derive(Debug, Clone)]
69pub enum Val {
70    Frame(Frame),
71    FrameType(FrameType),
72    Bool(bool),
73    Int(i64),
74    String(String),
75    Bytes(Vec<u8>),
76    Float(f64),
77    List(Vec<Val>),
78}

80#[derive(PartialEq, Eq, Debug, Clone)]
81pub struct FrameType {
82    pub width: usize,
83    pub height: usize,
84    pub format: ffi::AVPixelFormat,
85}
86
87impl FrameType {
88    pub fn new(width: usize, height: usize, format: ffi::AVPixelFormat) -> Self {
89        FrameType {
90            width,
91            height,
92            format,
93        }
94    }
95}
96
97impl Val {
98    pub fn from_expr(expr: &crate::sir::DataExpr, context: &crate::dve::Context) -> Self {
99        match expr {
100            crate::sir::DataExpr::Bool(b) => Val::Bool(*b),
101            crate::sir::DataExpr::Int(i) => Val::Int(*i),
102            crate::sir::DataExpr::String(s) => Val::String(s.to_string()),
103            crate::sir::DataExpr::Bytes(b) => Val::Bytes(b.clone()),
104            crate::sir::DataExpr::Float(f) => Val::Float(*f),
105            crate::sir::DataExpr::List(list) => {
106                let list = list
107                    .iter()
108                    .map(|expr| Val::from_expr(expr, context))
109                    .collect();
110                Val::List(list)
111            }
112        }
113    }
114
115    pub(crate) fn as_int(&self) -> Option<i64> {
116        match self {
117            Val::Int(i) => Some(*i),
118            _ => None,
119        }
120    }
121}

123/// A filter that can create a video frame from some inputs
124pub trait Filter: Send + Sync {
125    /// Creates a video frame from some inputs
126    ///
127    /// The inputs are arbitrary and user provided. An input can either be
128    /// ordered (arg) or named (kwargs). This is similar to Python function calls.
129    fn filter(
130        &self,
131        args: &[Val],
132        kwargs: &BTreeMap<String, Val>,
133    ) -> Result<Frame, crate::dve::Error>;
134
135    fn filter_type(
136        &self,
137        args: &[Val],
138        kwargs: &BTreeMap<String, Val>,
139    ) -> Result<FrameType, crate::dve::Error>;
140}

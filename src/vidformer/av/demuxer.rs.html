9struct IoCtx {
10    canary: u64, // We're doing some unsafe opaque pointer passing, so let's add a canary to make sure we didn't mess up. Valid value is 0xdeadbeef
11    size: u64,
12    reader: Box<dyn crate::io::ReadSeek>,
13    err: Option<std::io::Error>,
14}
15
16unsafe extern "C" fn vidformer_avio_read_packet(
17    opaque: *mut ::std::os::raw::c_void,
18    buf: *mut u8,
19    buf_size: ::std::os::raw::c_int,
20) -> ::std::os::raw::c_int {
21    let io_ctx = &mut *(opaque as *mut IoCtx);
22    debug_assert_eq!(io_ctx.canary, 0xdeadbeef);
23
24    let buf: &mut [u8] = unsafe { slice::from_raw_parts_mut(buf, buf_size as usize) };
25    let read = io_ctx.reader.read(buf);
26    match read {
27        Ok(read) => {
28            if read == 0 {
29                debug_assert!(io_ctx.reader.stream_position().unwrap() == io_ctx.size);
30                ffi::AVERROR_EOF
31            } else {
32                read as i32
33            }
34        }
35        Err(e) => {
36            error!("Error reading packet: {}", e);
37            if io_ctx.err.is_none() {
38                io_ctx.err = Some(e);
39            }
40            ffi::AVERROR_EXTERNAL
41        }
42    }
43}

45unsafe extern "C" fn vidformer_avio_seek(
46    opaque: *mut ::std::os::raw::c_void,
47    offset: i64,
48    whence: ::std::os::raw::c_int,
49) -> i64 {
50    let io_ctx = &mut *(opaque as *mut IoCtx);
51    debug_assert_eq!(io_ctx.canary, 0xdeadbeef);
52
53    let whence = match whence as u32 {
54        ffi::SEEK_CUR => std::io::SeekFrom::Current(offset),
55        ffi::SEEK_END => std::io::SeekFrom::End(offset),
56        ffi::SEEK_SET => std::io::SeekFrom::Start(offset as u64),
57        ffi::AVSEEK_SIZE => {
58            // whence == AVSEEK_SIZE is the libav way of asking for the size of the file
59            return io_ctx.size as i64;
60        }
61        ffi::AVSEEK_FORCE => panic!("AVSEEK_FORCE is not supported"), // libav way of saying "seek even if you have to reopen the file"
62        _ => panic!("invalid seek whence ({})", whence,),
63    };
64
65    let seeked = io_ctx.reader.seek(whence);
66    match seeked {
67        Ok(seeked) => seeked as i64,
68        Err(e) => {
69            error!("Error seeking: {}", e);
70            if io_ctx.err.is_none() {
71                io_ctx.err = Some(e);
72            }
73            ffi::AVERROR_EXTERNAL as i64
74        }
75    }
76}

78pub struct Demuxer {
79    pub format_context: *mut ffi::AVFormatContext,
80    avio_context: *mut ffi::AVIOContext,
81    #[allow(unused)] // We need to keep this alive since libav keeps it as an opaque pointer
82    io_ctx: std::pin::Pin<std::boxed::Box<IoCtx>>,
83    pub time_base: Rational64,
84    pub codec: *const ffi::AVCodec,
85    pub codec_parameters: *const ffi::AVCodecParameters,
86    pub video_stream_index: Option<usize>,
87    pub stream: *mut ffi::AVStream,
88}

90impl Demuxer {
91    pub fn new(
92        file_path: &str,
93        stream_idx: usize,
94        service: &crate::service::Service,
95        file_size: u64,
96        io_runtime_handle: &tokio::runtime::Handle,
97        io_cache: Option<(&dyn crate::io::IoWrapper, &str)>,
98    ) -> Result<Self, crate::Error> {
99        let mut format_context = unsafe { ffi::avformat_alloc_context() };
100        if format_context.is_null() {
101            return Err(crate::Error::AVError(
102                "could not allocate memory for Format Context".to_string(),
103            ));
104        }
105
106        debug!("Opening {} for read", file_path);
107
108        let op = service.blocking_operator(io_runtime_handle)?;
109
110        let reader: opendal::BlockingReader = op.reader(file_path).map_err(|e| {
111            if e.kind() == opendal::ErrorKind::NotFound {
112                crate::Error::IOError(format!("File `{}` not found", file_path))
113            } else {
114                crate::Error::IOError(format!("OpenDAL error: {}", e))
115            }
116        })?;
117
118        let reader: opendal::StdReader = match reader.into_std_read(0..file_size) {
119            Ok(reader) => reader,
120            Err(err) => {
121                return Err(crate::Error::IOError(format!(
122                    "OpenDAL failed to convert BlockingReader to StdReader: {}",
123                    err
124                )));
125            }
126        };
127
128        let reader = match io_cache {
129            Some((io_wrapper, fuid)) => io_wrapper.wrap(Box::new(reader), fuid),
130            None => Box::new(std::io::BufReader::with_capacity(128 * 1024, reader)),
131        };
132
133        let io_ctx = IoCtx {
134            canary: 0xdeadbeef,
135            size: file_size,
136            reader: Box::new(reader),
137            err: None,
138        };
139        let io_ctx = Box::pin(io_ctx);
140        let io_ctx_ptr =
141            io_ctx.as_ref().get_ref() as *const IoCtx as *mut IoCtx as *mut ::std::os::raw::c_void;
142
143        let avio_buffer_size = 16 * 1024;
144        let avio_buffer: *mut std::ffi::c_void =
145            unsafe { ffi::av_malloc(avio_buffer_size as usize) };
146
147        let avio_context = unsafe {
148            ffi::avio_alloc_context(
149                avio_buffer as *mut u8,
150                avio_buffer_size,
151                0,
152                io_ctx_ptr,
153                Some(vidformer_avio_read_packet),
154                None,
155                Some(vidformer_avio_seek),
156            )
157        };
158
159        if avio_context.is_null() {
160            return Err(crate::Error::AVError(
161                "could not allocate memory for AVIO Context".to_string(),
162            ));
163        }
164
165        unsafe {
166            (*format_context).pb = avio_context;
167        }
168
169        let ret = unsafe {
170            ffi::avformat_open_input(
171                &mut format_context,
172                ptr::null_mut(),
173                ptr::null_mut(),
174                ptr::null_mut(),
175            )
176        };
177        if ret != 0 {
178            if ret == ffi::AVERROR_EXTERNAL {
179                let err = io_ctx.err.as_ref().unwrap();
180                if err.kind() == std::io::ErrorKind::NotFound {
181                    return Err(crate::Error::IOError(format!(
182                        "File `{}` not found",
183                        file_path
184                    )));
185                } else {
186                    return Err(crate::Error::IOError(format!("OpenDAL error: {}", err)));
187                }
188            }
189
190            let err_str = crate::util::av_strerror(ret);
191            return Err(crate::Error::AVError(format!(
192                "failed to open media format: {}",
193                err_str
194            )));
195        }
196
197        // TODO: This may decode a few frames, which could be slow. Maybe don't do that when actually running a spec?
198        if unsafe { ffi::avformat_find_stream_info(format_context, ptr::null_mut()) } < 0 {
199            return Err(crate::Error::AVError(
200                "could not get the stream info".to_string(),
201            ));
202        }
203
204        let mut codec_ptr: *const ffi::AVCodec = ptr::null_mut();
205        let mut codec_parameters_ptr: *const ffi::AVCodecParameters = ptr::null_mut();
206        let mut video_stream_index = None;
207
208        let streams = unsafe {
209            slice::from_raw_parts_mut(
210                (*format_context).streams,
211                (*format_context).nb_streams as usize,
212            )
213        };
214
215        for (i, stream) in streams
216            .iter_mut()
217            .map(|stream| unsafe { stream.as_mut() }.expect("null stream pointer"))
218            .enumerate()
219        {
220            if i != stream_idx {
221                stream.discard = ffi::AVDiscard_AVDISCARD_ALL;
222                continue;
223            }
224
225            let local_codec_params = unsafe { stream.codecpar.as_ref() }.expect("codecpar is null");
226            let local_codec =
227                unsafe { ffi::avcodec_find_decoder(local_codec_params.codec_id).as_ref() }
228                    .expect("ERROR unsupported codec!");
229
230            if local_codec_params.codec_type == ffi::AVMediaType_AVMEDIA_TYPE_VIDEO {
231                if video_stream_index.is_none() {
232                    video_stream_index = Some(i);
233                    codec_ptr = local_codec;
234                    codec_parameters_ptr = local_codec_params;
235                }
236            } else {
237                // The requested stream is not a video stream
238                let codec_type = match local_codec_params.codec_type {
239                    ffi::AVMediaType_AVMEDIA_TYPE_AUDIO => "audio",
240                    ffi::AVMediaType_AVMEDIA_TYPE_SUBTITLE => "subtitle",
241                    ffi::AVMediaType_AVMEDIA_TYPE_DATA => "data",
242                    ffi::AVMediaType_AVMEDIA_TYPE_ATTACHMENT => "attachment",
243                    _ => "unknown",
244                };
245                return Err(crate::Error::AVError(format!(
246                    "Stream {} is not a video stream (it is a {} stream). Please specify a valid video stream index.",
247                    i, codec_type
248                )));
249            }
250        }
251
252        let time_base = crate::util::avrat_to_rat(&unsafe { (*streams[stream_idx]).time_base });
253        debug_assert!(!time_base.is_zero());
254
255        Ok(Demuxer {
256            format_context,
257            avio_context,
258            io_ctx,
259            time_base,
260            codec: codec_ptr,
261            codec_parameters: codec_parameters_ptr,
262            video_stream_index,
263            stream: unsafe { streams[stream_idx].as_mut() }.unwrap(),
264        })
265    }
266
267    pub fn seek(&mut self, ts: &Rational64) -> Result<(), crate::Error> {
268        let seek_ts = ts / self.time_base;
269        assert_eq!(*seek_ts.denom(), 1);
270        let seek_ts = *seek_ts.numer();
271
272        debug!("Seeking to {}", ts);
273
274        if unsafe { ffi::avformat_seek_file(self.format_context, 0, seek_ts, seek_ts, seek_ts, 0) }
275            < 0
276        {
277            return Err(crate::Error::AVError("failed to seek file".to_string()));
278        }
279
280        Ok(())
281    }

283    pub fn read_packet(&mut self, packet: *mut ffi::AVPacket) -> Option<()> {
284        loop {
285            if unsafe { ffi::av_read_frame(self.format_context, packet) } >= 0 {
286                // debug!(
287                //     "DEMUX - Packet stream index {} pts = {} dts = {} key_frame = {} size = {}",
288                //     unsafe { (*packet).stream_index },
289                //     unsafe { (*packet).pts },
290                //     unsafe { (*packet).dts },
291                //     unsafe { (*packet).flags } as u32 & ffi::AV_PKT_FLAG_KEY != 0,
292                //     unsafe { (*packet).size }
293                // );
294
295                // Is this packet from the correct stream?
296                if unsafe { (*packet).stream_index } as usize == self.video_stream_index.unwrap() {
297                    // If so, we have updated the pointer, so return
298                    return Some(());
299                } else {
300                    // If not, unref the packet and keep trying
301                    unsafe {
302                        ffi::av_packet_unref(packet);
303                    }
304                    continue;
305                }
306            } else {
307                return None;
308            }
309        }
310    }
311
312    pub fn close(&self) {
313        unsafe {
314            ffi::avformat_close_input(&mut (self.format_context as *mut _));
315
316            // Free the avio buffer. libav and us share memory control over it, so its probably not even the one we allocated.
317            // Sorry that pointer casting is so ugly, it makes sure to pass the pointer to the buffer pointer, not the buffer pointer itself.
318            // Probably a better way to do this which doesn't involve using the mut keyword six times in one expression. ¯\_(ツ)_/¯
319            ffi::av_freep(
320                &mut (*self.avio_context).buffer as *mut *mut u8 as *mut *mut std::os::raw::c_void
321                    as *mut std::os::raw::c_void,
322            );
323
324            ffi::av_free(self.avio_context as *mut std::os::raw::c_void);
325        }
326    }

338    fn tos_test() {
339        let service = crate::service::Service::default();
340        let profile = crate::source::SourceVideoStreamMeta::profile(
341            "tos_720p",
342            "../tos_720p.mp4",
343            0,
344            &service,
345            None,
346        )
347        .unwrap();
348
349        let io_runtime = tokio::runtime::Builder::new_multi_thread()
350            .worker_threads(1)
351            .enable_all()
352            .build()
353            .unwrap();
354
355        let mut demuxer = Demuxer::new(
356            TEST_VID,
357            TEST_STREAM,
358            &service,
359            profile.file_size,
360            io_runtime.handle(),
361            None,
362        )
363        .unwrap();
364
365        let packet = unsafe { ffi::av_packet_alloc().as_mut() }
366            .expect("failed to allocated memory for AVPacket");
367
368        let mut demuxed_packets = 0;
369        while demuxer.read_packet(packet).is_some() {
370            demuxed_packets += 1;
371        }
372        demuxer.close();
373
374        assert_eq!(demuxed_packets, 17616);
375    }

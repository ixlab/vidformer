1use crate::dve::AVFrame;
2use crate::util;
3use log::*;
4use num_rational::Rational64;
5use rusty_ffmpeg::ffi;
6use std::ffi::CString;
7use std::ptr;
8
9use crate::dve::Config;
10
11pub struct Encoder {
12    pub(crate) codec_ctx: *mut ffi::AVCodecContext,
13    pub(crate) packet: *mut ffi::AVPacket,
14    pub(crate) time_base: Rational64,
15    pub(crate) flushed: bool,
16}
17
18impl Encoder {
19    pub(crate) fn new(config: &Config, time_base: &Rational64) -> Result<Self, crate::Error> {
20        let codec = match &config.encoder {
21            Some(enc_cfg) => enc_cfg.avcodec()?,
22            None => {
23                let codec = unsafe { ffi::avcodec_find_encoder(ffi::AVCodecID_AV_CODEC_ID_H264) };
24                match unsafe { codec.as_ref() } {
25                    Some(codec) => codec,
26                    None => panic!("Failed to find default h264 encoder"),
27                }
28            }
29        };
30
31        debug!("Encoder has codec {}", util::fmt_av_codec(codec));
32
33        let codec_ctx: *mut ffi::AVCodecContext = unsafe { ffi::avcodec_alloc_context3(codec) };
34        if codec_ctx.is_null() {
35            return Err(crate::Error::AVError(
36                "Failed to allocate codec context".to_string(),
37            ));
38        }
39
40        let default_opts = &[("preset".to_string(), "ultrafast".to_string())];
41
42        let opts: &[(String, String)] = match &config.encoder {
43            Some(enc_cfg) => &enc_cfg.opts,
44            None => default_opts,
45        };
46
47        for (opt_k, opt_v) in opts {
48            let opt_k_cstr = CString::new(opt_k.clone()).unwrap();
49            let opt_v_cstr = CString::new(opt_v.clone()).unwrap();
50
51            let ret = unsafe {
52                ffi::av_opt_set(
53                    (*codec_ctx).priv_data,
54                    opt_k_cstr.as_ptr(),
55                    opt_v_cstr.as_ptr(),
56                    0,
57                )
58            };
59            if ret < 0 {
60                return Err(crate::Error::AVError(format!(
61                    "Failed to set encoder opt `{}` to `{}`",
62                    opt_k, opt_v
63                )));
64            }
65        }
66
67        let pix_fmt_name = CString::new(config.output_pix_fmt.clone()).unwrap();
68        let output_pix_fmt = unsafe { ffi::av_get_pix_fmt(pix_fmt_name.as_ptr()) };
69        if output_pix_fmt == ffi::AVPixelFormat_AV_PIX_FMT_NONE {
70            return Err(crate::Error::ConfigError(format!(
71                "Failed to find output pix fmt `{}`",
72                config.output_pix_fmt
73            )));
74        }
75
76        // If supported pix_fmts are known make sure the output pix_fmt is supported
77        if !codec.pix_fmts.is_null() {
78            let mut found = false;
79            for i in 0.. {
80                let pix_fmt = unsafe { *codec.pix_fmts.add(i) };
81                if pix_fmt == -1 {
82                    break;
83                }
84                if pix_fmt == output_pix_fmt {
85                    found = true;
86                    break;
87                }
88            }
89            if !found {
90                return Err(crate::Error::ConfigError(format!(
91                    "Output pix_fmt `{}` not supported by encoder",
92                    config.output_pix_fmt
93                )));
94            }
95        }
96
97        let av_time_base = crate::util::rat_to_avrat(time_base);
98        unsafe {
99            (*codec_ctx).height = config.output_height as i32;
100            (*codec_ctx).width = config.output_width as i32;
101            (*codec_ctx).time_base = av_time_base;
102            (*codec_ctx).gop_size = 10; // TODO: Don't set?
103            (*codec_ctx).pix_fmt = output_pix_fmt;
104            (*codec_ctx).flags |= ffi::AV_CODEC_FLAG_GLOBAL_HEADER as i32;
105        }
106
107        if unsafe { ffi::avcodec_open2(codec_ctx, codec, ptr::null_mut()) } < 0 {
108            return Err(crate::Error::AVError("Failed to open encoder".to_string()));
109        }
110
111        let packet = unsafe { ffi::av_packet_alloc() };
112        if packet.is_null() {
113            return Err(crate::Error::AVError(
114                "Failed to allocate packet".to_string(),
115            ));
116        }
117
118        Ok(Encoder {
119            codec_ctx,
120            packet,
121            time_base: *time_base,
122            flushed: false,
123        })
124    }
125
126    pub(crate) fn encode(&mut self, pts: &Rational64, frame: &AVFrame) -> Result<(), crate::Error> {
127        // TODO: Do this elsewhere?
128        if unsafe { ffi::av_frame_make_writable(frame.inner) } < 0 {
129            return Err(crate::Error::AVError(
130                "Failed to make frame writable".to_string(),
131            ));
132        }
133        unsafe {
134            // It's none of our business what the input frame type is
135            // Also, we don't want the encoder to complain if something looks weird
136            (*frame.inner).pict_type = ffi::AVPictureType_AV_PICTURE_TYPE_NONE;
137        }
138
139        let time_scaled = pts / self.time_base;
140        assert!(*time_scaled.denom() == 1);
141        debug!("ENCODE - Set packet pts to {}", *time_scaled.numer());
142        unsafe {
143            (*frame.inner).pts = *time_scaled.numer();
144            (*frame.inner).pkt_dts = 0;
145        }
146
147        let ret = unsafe { ffi::avcodec_send_frame(self.codec_ctx, frame.inner) };
148        if ret < 0 {
149            let error = ffi::av_err2str(ret);
150            return Err(crate::Error::AVError(format!(
151                "Failed to send frame to encoder: {}",
152                error
153            )));
154        }
155
156        Ok(())
157    }
158
159    pub(crate) fn flush(&mut self) -> Result<(), crate::Error> {
160        assert!(!self.flushed);
161        self.flushed = true;
162        if unsafe { ffi::avcodec_send_frame(self.codec_ctx, ptr::null_mut()) } < 0 {
163            return Err(crate::Error::AVError("Failed to flush encoder".to_string()));
164        }
165
166        Ok(())
167    }
168
169    pub(crate) fn get_packet(&mut self) -> Option<*mut ffi::AVPacket> {
170        if unsafe { ffi::avcodec_receive_packet(self.codec_ctx, self.packet) } >= 0 {
171            Some(self.packet)
172        } else {
173            None
174        }
175    }
176
177    pub(crate) fn close(&mut self) {
178        assert!(self.flushed);
179        assert!(self.get_packet().is_none());
180
181        info!("Closing encoder");
182        unsafe {
183            ffi::av_packet_free(&mut self.packet);
184            ffi::avcodec_free_context(&mut self.codec_ctx);
185        }
186    }
187}

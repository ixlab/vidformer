1//! Specs declarativly define an edited video.
2//!
3//! A spec defines an output video (as a logical time-indexed array).
4//! It creates this array by defining its timestamps (the set of times at which the spec is defined) and a function that maps each time to a frame.
5//! The frame is represented as an [`FrameExpr`] expression.
6//!
7//! Specs are stateless and immutable. They are intended to be smaller than just storing the entire output video as an array.
8//! Additionally, this generic spec interface allows the general case of a video-editing DSL, as done in V2V, while allowing for whatever language the use case needs.
9
10use crate::sir::FrameExpr;
11use num_rational::Rational64;
12
13/// A trait for providing information to a spec during runtime
14pub trait SpecContext {}
15
16/// A trait for a spec
17///
18/// A spec defines a video from a sequence of transformations.
19/// A [`SpecContext`] is provided for information about the available source videos and arrays.
20pub trait Spec: Sync + Send {
21    /// Returns the timestamps of the outupt video.
22    ///
23    /// The timestamps are the set of times at which the spec is defined.
24    /// Each time corresponds to a single output frame at that timestamp.
25    ///
26    /// The output must:
27    /// - Be sorted in ascending order
28    /// - Contain no duplicate values
29    /// - Begin with 0
30    fn timestamps(&self, context: &dyn SpecContext) -> Vec<Rational64>;
31
32    /// Returns the "logically" rendered frame at a given time.
33    ///
34    /// This function should assume that the time is in the domain of the spec.
35    fn render(&self, context: &dyn SpecContext, t: &Rational64) -> FrameExpr;
36}
37
38pub(crate) fn get_framerate(spec: &dyn Spec) -> usize {
39    // TODO: Not all framerates are widely supported in HLS. We should probably add a check for that.
40
41    let context = crate::dve::EmptySpecCtx;
42    let frame_times = spec.timestamps(&context);
43
44    let mut frame_deltas = frame_times.windows(2).map(|w| w[1] - w[0]);
45
46    // Default to 30fps if there's only one frame. Not ideal, but framerate doesn't matter in that case.
47    // We don't want a user to try one frame and it break on them.
48    // TODO: Error on this case?
49    let first_delta = frame_deltas.next().unwrap_or(Rational64::new(1, 30));
50
51    assert!(frame_deltas.all(|d| d == first_delta));
52    assert_eq!(*first_delta.numer(), 1, "Frame rate must be an integer");
53
54    *first_delta.denom() as usize
55}

57/// A spec backed by a JSON file
58#[derive(Debug, serde::Deserialize, serde::Serialize)]
59pub struct JsonSpec {
60    pub frames: Vec<(Rational64, FrameExpr)>,
61}
62
63impl Spec for JsonSpec {
64    fn timestamps(&self, _context: &dyn SpecContext) -> Vec<Rational64> {
65        self.frames.iter().map(|(t, _)| *t).collect()
66    }
67
68    fn render(&self, _context: &dyn SpecContext, t: &Rational64) -> FrameExpr {
69        let index = self
70            .frames
71            .binary_search_by(|(t2, _)| t2.partial_cmp(t).unwrap())
72            .unwrap();
73
74        self.frames[index].1.clone()
75    }
76}

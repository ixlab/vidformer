27    pub(crate) fn new(
28        iframes_per_oframe: Vec<BTreeSet<IFrameRef>>,
29        iframe_refs_in_out_idx: BTreeMap<IFrameRef, BTreeSet<usize>>,
30        dve_context: Arc<Context>,
31        dve_config: Arc<Config>,
32    ) -> Result<Self, crate::Error> {
33        if dve_config.decode_pool_size == 0 {
34            return Err(crate::Error::ConfigError(
35                "decode_pool_size must be greater than 0".to_string(),
36            ));
37        }
38
39        let mut out = Pool {
40            done_gens_recent: BTreeSet::new(),
41            done_gens_past: 0,
42            next_gen: 0,
43            members: BTreeMap::new(),
44            decoders: BTreeMap::new(),
45            finished_unjoined_decoders: BTreeSet::new(),
46            terminate_decoders: false,
47            iframes_per_oframe,
48            iframe_refs_in_out_idx,
49            dve_context,
50            dve_config,
51        };
52        while out.plan_gen() {}
53        Ok(out)
54    }

56    fn next_needed_gen(&self, frame: &IFrameRef) -> usize {
57        let frame_uses = match self.iframe_refs_in_out_idx.get(frame) {
58            Some(uses) => uses,
59            None => return F_NOT_USED,
60        };
61        // TODO: We could throw a binary search in here to speed up cases where there are many uses of a frame, such as a watermark image
62        for frame_use_gen in frame_uses {
63            if frame_use_gen >= &self.done_gens_past
64                && !self.done_gens_recent.contains(frame_use_gen)
65            {
66                return *frame_use_gen;
67            }
68        }
69        F_NOT_USED
70    }

72    fn decoder_next_needed_gen(&self, decoder_id: &str) -> usize {
73        let decoder = self.decoders.get(decoder_id).unwrap();
74        let mut next_needed_gen = F_NOT_USED;
75        for frame in &decoder.future_frames {
76            let frame_next_needed = self.next_needed_gen(&IFrameRef {
77                sourceref: decoder.source.clone(),
78                pts: *frame,
79            });
80            if frame_next_needed < next_needed_gen {
81                next_needed_gen = frame_next_needed;
82            }
83        }
84        next_needed_gen
85    }

87    fn frame_gop(&self, frame: &IFrameRef) -> usize {
88        let source = self.dve_context.sources.get(&frame.sourceref).unwrap();
89        debug_assert!(source.ts.binary_search(&frame.pts).is_ok());
90        match source.keys.binary_search(&frame.pts) {
91            Ok(i) => i,
92            Err(i) => i - 1, // i is the index of the first element greater than frame.pts
93        }
94    }

96    pub(crate) fn new_decoder_gop(&self) -> Option<(SourceRef, usize)> {
97        debug_assert!(self.decoders.len() <= self.dve_config.decoders);
98        if self.decoders.len() == self.dve_config.decoders {
99            return None;
100        }
101
102        let mut soonest_needed_basis_frame: Option<&IFrameRef> = None;
103        let mut soonest_needed_basis_frame_next_needed = F_NOT_USED;
104
105        for frame in self.need_set() {
106            // Skip frames already in pool or being decoded; only consider basis frames
107            if self.members.contains_key(frame) || self.is_in_future_set(frame) {
108                continue;
109            }
110            let candidate_frame_next_needed = self.next_needed_gen(frame);
111            if candidate_frame_next_needed < soonest_needed_basis_frame_next_needed {
112                soonest_needed_basis_frame = Some(frame);
113                soonest_needed_basis_frame_next_needed = candidate_frame_next_needed;
114            }
115        }
116
117        soonest_needed_basis_frame.map(|frame| {
118            let gop_id = self.frame_gop(frame);
119            (frame.sourceref.clone(), gop_id)
120        })
121    }

123    fn eviction_set(
124        &self,
125        size: usize,
126        next_need_set: &BTreeSet<IFrameRef>,
127    ) -> BTreeSet<IFrameRef> {
128        struct FrameEvictionCandidate<'b> {
129            needed_gen: usize,
130            frame_ts: &'b IFrameRef,
131        }
132        impl Ord for FrameEvictionCandidate<'_> {
133            fn cmp(&self, other: &Self) -> std::cmp::Ordering {
134                self.needed_gen.cmp(&other.needed_gen)
135            }
136        }
137        impl PartialOrd for FrameEvictionCandidate<'_> {
138            fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
139                Some(self.cmp(other))
140            }
141        }
142        impl Eq for FrameEvictionCandidate<'_> {}
143        impl PartialEq for FrameEvictionCandidate<'_> {
144            fn eq(&self, other: &Self) -> bool {
145                self.needed_gen == other.needed_gen
146            }
147        }
148
149        let mut heap = std::collections::BinaryHeap::new();
150        for frame_ts in self.members.keys() {
151            if !next_need_set.contains(frame_ts) {
152                heap.push(FrameEvictionCandidate {
153                    needed_gen: self.next_needed_gen(frame_ts),
154                    frame_ts,
155                });
156            }
157        }
158
159        debug_assert!(heap.len() >= size);
160        let mut out = BTreeSet::new();
161        for _ in 0..size {
162            let evicted = heap.pop().unwrap();
163            out.insert(evicted.frame_ts.clone());
164        }
165        out
166    }

168    pub(crate) fn should_stall(&self, decoder_id: &str) -> bool {
169        let decoder = self.decoders.get(decoder_id).unwrap();
170
171        for pts in &decoder.future_frames {
172            let iframe_ref = IFrameRef {
173                sourceref: decoder.source.clone(),
174                pts: *pts,
175            };
176
177            if self.members.contains_key(&iframe_ref) {
178                continue;
179            }
180
181            if let Some(frame_uses) = self.iframe_refs_in_out_idx.get(&iframe_ref) {
182                for gen in frame_uses {
183                    if *gen >= self.done_gens_past
184                        && *gen < self.next_gen
185                        && !self.done_gens_recent.contains(gen)
186                    {
187                        return false;
188                    }
189                }
190            }
191        }
192
193        true
194    }

196    pub(crate) fn decoded(&mut self, decoder_id: &str, frame: IFrameRef, avframe: Arc<AVFrame>) {
197        debug_assert!(self.members.len() <= self.dve_config.decode_pool_size);
198        debug_assert!(!self.should_stall(decoder_id));
199
200        if self.members.contains_key(&frame) {
201            // Frame already present
202            return;
203        }
204
205        let need_set: BTreeSet<IFrameRef> = self.need_set().into_iter().cloned().collect();
206
207        if need_set.contains(&frame) || self.members.len() < self.dve_config.decode_pool_size {
208            // If the pool is full evict a cache frame
209            if self.members.len() == self.dve_config.decode_pool_size {
210                let evict_set = self.eviction_set(1, &need_set);
211                debug_assert_eq!(evict_set.len(), 1);
212                for frame_ts in evict_set {
213                    self.members.remove(&frame_ts);
214                }
215            }
216            self.members.insert(frame, avframe);
217        } else {
218            // See if we can evict a frame to make room
219            let f_next_need = self.next_needed_gen(&frame);
220            if f_next_need < F_NOT_USED {
221                let mut least_needed_pool_frame: Option<IFrameRef> = None;
222                let mut least_needed_pool_frame_next_needed = F_NOT_USED;
223
224                for pool_frame in self.members.keys() {
225                    if !need_set.contains(pool_frame) {
226                        if least_needed_pool_frame.is_some() {
227                            let pool_frame_next_needed = self.next_needed_gen(pool_frame);
228                            if pool_frame_next_needed > least_needed_pool_frame_next_needed {
229                                least_needed_pool_frame = Some(pool_frame.clone());
230                                least_needed_pool_frame_next_needed = pool_frame_next_needed;
231                            }
232                        } else {
233                            least_needed_pool_frame = Some(pool_frame.clone());
234                            least_needed_pool_frame_next_needed = self.next_needed_gen(pool_frame);
235                        }
236                    }
237                }
238
239                if f_next_need < least_needed_pool_frame_next_needed {
240                    log::info!("Evicting frame {:?} (needed in gen {}) for sooner-needed frame {:?} (needed in gen {})", least_needed_pool_frame, least_needed_pool_frame_next_needed, frame, f_next_need);
241                    self.members.remove(&least_needed_pool_frame.unwrap());
242                    self.members.insert(frame, avframe);
243                }
244            }
245        }
246
247        debug_assert!(self.members.len() <= self.dve_config.decode_pool_size);
248    }

250    pub(crate) fn should_decoder_abandon(&self, decoder_id: &str) -> bool {
251        if self.decoders.len() < self.dve_config.decoders || !self.should_stall(decoder_id) {
252            return false;
253        }
254
255        let dec_next_needed_gen = self.decoder_next_needed_gen(decoder_id);
256
257        let mut found_sooner_basis_frame = false;
258        for frame in self.need_set() {
259            if self.members.contains_key(frame) || self.is_in_future_set(frame) {
260                continue;
261            }
262            let frame_next_needed_gen = self.next_needed_gen(frame);
263            if frame_next_needed_gen < dec_next_needed_gen {
264                found_sooner_basis_frame = true;
265                break;
266            }
267        }
268        if !found_sooner_basis_frame {
269            return false;
270        }
271
272        // check if all other decoders have a lower or equal next needed gen
273        // aka make sure this is the least-soonest-needed decoder
274        for other_decoder_id in self.decoders.keys() {
275            if other_decoder_id == decoder_id {
276                continue;
277            }
278            let other_dec_next_needed_gen = self.decoder_next_needed_gen(other_decoder_id);
279            if other_dec_next_needed_gen > dec_next_needed_gen {
280                return false;
281            }
282        }
283
284        true
285    }

287    fn is_in_future_set(&self, frame: &IFrameRef) -> bool {
288        self.decoders.values().any(|decoder| {
289            decoder.source == frame.sourceref && decoder.future_frames.contains(&frame.pts)
290        })
291    }

293    fn plan_gen(&mut self) -> bool {
294        debug_assert!(self.next_gen <= self.iframes_per_oframe.len());
295        if self.next_gen == self.iframes_per_oframe.len() {
296            return false;
297        }
298
299        let number_of_active_gens =
300            self.next_gen - self.done_gens_past - self.done_gens_recent.len();
301        debug_assert_eq!(number_of_active_gens, self.active_gens().len());
302        if number_of_active_gens >= self.dve_config.decoder_view {
303            return false;
304        }
305
306        let mut next_need_set: BTreeSet<IFrameRef> =
307            { self.need_set().iter().map(|a| (*a).clone()).collect() };
308        next_need_set.extend(
309            self.iframes_per_oframe[self.next_gen]
310                .iter()
311                .map(|a| (*a).clone()),
312        );
313
314        if next_need_set.len() > self.dve_config.decode_pool_size {
315            // not enough space in the pool even with evictions
316            return false;
317        }
318
319        // Enough space, but check if we need to evict some frames first
320        let members_not_in_need_set = self
321            .members
322            .keys()
323            .filter(|k| !next_need_set.contains(*k))
324            .count();
325        let union_size = next_need_set.len() + members_not_in_need_set;
326        if union_size > self.dve_config.decode_pool_size {
327            let needed_evictions = union_size - self.dve_config.decode_pool_size;
328            let evict_set = self.eviction_set(needed_evictions, &next_need_set);
329            debug_assert!(evict_set.len() == needed_evictions);
330            for frame_ts in evict_set {
331                debug_assert!(!next_need_set.contains(&frame_ts));
332                self.members.remove(&frame_ts);
333            }
334        }
335        self.next_gen += 1;
336        true
337    }

339    pub(crate) fn active_gens(&self) -> BTreeSet<usize> {
340        let mut out = BTreeSet::new();
341        for g in self.done_gens_past..self.next_gen {
342            if !self.done_gens_recent.contains(&g) {
343                out.insert(g);
344            }
345        }
346        out
347    }

349    pub(crate) fn finish_gen(&mut self, gen: usize) {
350        debug_assert!(self.active_gens().contains(&gen));
351
352        // update the done_gens
353        self.done_gens_recent.insert(gen);
354        loop {
355            match self.done_gens_recent.first() {
356                Some(&first) if first == self.done_gens_past => {
357                    self.done_gens_recent.remove(&first);
358                    self.done_gens_past += 1;
359                }
360                _ => break,
361            }
362        }
363
364        // plan future gens
365        while self.plan_gen() {}
366    }

368    pub(crate) fn is_gen_ready(&self, gen: usize) -> bool {
369        debug_assert!(self.active_gens().contains(&gen));
370
371        self.iframes_per_oframe[gen]
372            .iter()
373            .all(|dep_frame| self.members.contains_key(dep_frame))
374    }

376    pub(crate) fn get_ready_gen_frames(&self, gen: usize) -> BTreeMap<IFrameRef, Arc<AVFrame>> {
377        debug_assert!(self.is_gen_ready(gen));
378        self.iframes_per_oframe[gen]
379            .iter()
380            .map(|iframeref| (iframeref.clone(), self.members[iframeref].clone()))
381            .collect()
382    }

384    pub(crate) fn need_set(&self) -> BTreeSet<&IFrameRef> {
385        let mut out = BTreeSet::new();
386        for gen in self.done_gens_past..self.next_gen {
387            if !self.done_gens_recent.contains(&gen) {
388                out.extend(self.iframes_per_oframe[gen].iter());
389            }
390        }
391        out
392    }
